 5 /**
 6  * Expands the body classes added by WordPress. <br />
 7  * Only called by `add_filter('body_class','d7_body_classes');`
 8  *
 9  * ### Added classes
10  * * `.post_type-{post_type}`
11  * * `.post_name-{post_name}`
12  * * `.taxonomy-{tax_name}`
13  * * `.taxonomy_term-{tax_term_slug}`
14  * * `.taxonomy_id-{tax_id}`
15  * * `.taxonomy_term_id-{tax_term_id}`
16  *
17  * @package d7
18  * @subpackage boilerplate-theme_filters+hooks
19  *
20  * @internal Called by `body_class` filter
21  *
22  */
23 function d7_body_classes($classes, $class='') {
24     global $wp_query;
25 
26     if ( isset($wp_query->queried_object) ) {
27 
28         // Post type
29         if ( isset($wp_query->queried_object->post_type) ) {
30             $classes[] = 'post_type-' . $wp_query->queried_object->post_type;
31         }
32 
33         // Post name
34         if ( isset($wp_query->queried_object->post_name) ) {
35             $classes[] = 'post_name-' . $wp_query->queried_object->post_name;
36         }
37 
38         // Taxonomy
39         if ( isset($wp_query->queried_object->taxonomy) ) {
40             $classes[] = 'taxonomy-' . $wp_query->queried_object->taxonomy;
41         }
42 
43         // Taxonomy term
44         if ( isset($wp_query->queried_object->taxonomy) ) {
45             $classes[] = 'taxonomy_term-' . $wp_query->queried_object->slug;
46         }
47 
48         // Taxonomy ID
49         if ( isset($wp_query->queried_object->cat_ID) ) {
50             $classes[] = 'taxonomy_id-' . $wp_query->queried_object->cat_ID;
51         }
52 
53         // Taxonomy term ID
54         if ( isset($wp_query->queried_object->term_id) ) {
55             $classes[] = 'taxonomy_term_id-' . $wp_query->queried_object->term_id;
56         }
57 
58     } // if isset
59 
60     // Has image
61     if ( has_post_thumbnail() ) {
62         // Using a dash instead of underscore because WP turns it
63         // into that anyway and I want to be more transparent
64         $classes[] = 'has-post-thumbnail';
65     }
66 
67     // Has comments or not
68     if ( comments_open() && get_comments_number() ) {
69         $classes[] = "has-comments";
70     } else {
71         $classes[] = "no-comments";
72     }
73 
74     // Comments open/closed
75     if ( comments_open() ) {
76         $classes[] = "can-comment";
77     }
78 
79     // Classes for sidebars
80     $classes[] = d7_sidebar_classes();
81 
82     return $classes;// return the $classes array
83 }
