 11 /**
 12  * Enqueue jQuery from Google CDN with fallback to local WordPress
 13  * First we have a really complex method of loading jQuery from the CDN or local
 14  * should the CDN fail. The code is adapted from https://gist.github.com/wpsmith/4083811.
 15  * Then we load basic styles.
 16  *
 17  * @package d7
 18  * @subpackage boilerplate-theme_filters+hooks
 19  * @internal only called by `wp_enqueue_scripts` action
 20  *
 21  * @link https://gist.github.com/wpsmith/4083811.
 22  * @link http://codex.wordpress.org/Function_Reference/wp_enqueue_script
 23  * @link http://codex.wordpress.org/Function_Reference/wp_register_script
 24  * @link http://codex.wordpress.org/Function_Reference/wp_deregister_script
 25  * @link http://codex.wordpress.org/Function_Reference/get_bloginfo
 26  * @link http://codex.wordpress.org/Function_Reference/is_wp_error
 27  * @link http://codex.wordpress.org/Function_Reference/set_transient
 28  * @link http://codex.wordpress.org/Function_Reference/get_transient
 29  *
 30  * @uses get_transient()        Get the value of a transient.
 31  * @uses set_transient()        Set/update the value of a transient.
 32  * @uses is_wp_error()          Check whether the passed variable is a WordPress Error.
 33  * @uses get_bloginfo()         returns information about your site.
 34  * @uses wp_deregister_script() Deregisters javascripts for use with wp_enqueue_script() later.
 35  * @uses wp_register_script()   Registers javascripts for use with wp_enqueue_script() later.
 36  * @uses wp_enqueue_script()    Enqueues javascript.
 37  */
 38 function d7_enqueue_jquery() {
 39     // Setup Google URI, default
 40     $protocol = ( isset( $_SERVER['HTTPS'] ) && 'on' == $_SERVER['HTTPS'] ) ? 'https' : 'http';
 41     // Get Latest Version
 42     $url = $protocol . '://code.jquery.com/jquery-1.11.1.min.js';
 43 
 44     // Get Specific Version
 45     //$url      = $protocol . '://ajax.googleapis.com/ajax/libs/jquery/1.8.2/jquery.min.js';
 46 
 47     // Setup WordPress URI
 48     $wpurl =  get_bloginfo('template_directory') . '/js/jquery-1.11.1.min.js';
 49 
 50     // Setup version
 51     $ver = null;
 52 
 53     // Deregister WordPress default jQuery
 54     wp_deregister_script( 'jquery' );
 55 
 56     // Check transient, if false, set URI to WordPress URI
 57     delete_transient( 'google_jquery' );
 58 
 59     if ( 'false' == ( $google = get_transient( 'google_jquery' ) ) ) {
 60         $url = $wpurl;
 61     }
 62     // Transient failed
 63     elseif ( false === $google ) {
 64         // Ping Google
 65         $resp = wp_remote_head( $url );
 66 
 67         // Use Google jQuery
 68         if ( ! is_wp_error( $resp ) && 200 == $resp['response']['code'] ) {
 69             // Set transient for 5 minutes
 70             set_transient( 'google_jquery', 'true', 60 * 5 );
 71         }
 72 
 73         // Use WordPress jQuery
 74         else {
 75             // Set transient for 5 minutes
 76             set_transient( 'google_jquery', 'false', 60 * 5 );
 77 
 78             // Use WordPress URI
 79             $url = $wpurl;
 80 
 81             // Set jQuery Version, WP stanards
 82             $ver = '1.8.2';
 83         }
 84     }
 85 
 86     // Register surefire jQuery
 87     wp_register_script( 'jquery', $url, array(), $ver, true );
 88 
 89     // Enqueue jQuery
 90     wp_enqueue_script( 'jquery' );
 91 
 92     // Now load basic site js
 93     wp_enqueue_script('basic', get_bloginfo('template_directory').'/js/main.js', array('jquery'), '1.0');
 94 
 95     // For comment reply form
 96     if ( is_singular() && get_option( 'thread_comments' ) && comments_open() ) {
 97         wp_enqueue_script( 'comment-reply' );
 98     }
 99 
100 }
