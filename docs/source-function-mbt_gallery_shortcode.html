  3 /**
  4  * Rewrite the markup for the gallery shortcode so that all gallery images
  5  * can open with fresco.js. The link to attachment page is not respected,
  6  * and size/column classes are added.
  7  *
  8  * @package d7
  9  * @subpackage boilerplate-theme_filters+hooks
 10  *
 11  * @internal only called by `post_gallery` filter
 12  *
 13  */
 14 function mbt_gallery_shortcode( $output = '', $atts, $content = false, $tag = false ) {
 15 
 16     // Get thumbnail size
 17     $size = isset($atts['size']) ? $atts['size'] : 'thumbnail';
 18 
 19     // Get column count. 3 is the default
 20     $columns = isset($atts['columns']) ? $atts['columns'] : 3;
 21 
 22     $output = '<div class="gallery galery--size-' . $size . ' gallery--columns-' .$columns . '">';
 23 
 24         // Load each item
 25         $ids = explode(",", $atts['include']);
 26         foreach ( $ids as $id ) {
 27 
 28 
 29             $image_large = wp_get_attachment_image_src($id, 'large');
 30 
 31             $image = wp_get_attachment_image($id, $size);
 32             $output .= '<figure class="gallery-item galery-item--size-' . $size . ' galery-item--columns-' . $columns . '">';
 33 
 34             // Unless link is set to none, link it!
 35             if ( $atts['link'] !== 'none' ) {
 36                 $output .= '<a class="fresco" data-fresco-group="' . get_the_ID() . '" href="' . $image_large[0] . '">';
 37             }
 38 
 39             // Image actual
 40             $output .= $image;
 41 
 42             // Close link if we're linking
 43             if ( $atts['link'] !== 'none' ) {
 44                 $output .= '</a>';
 45             }
 46 
 47             // Close figure
 48             $output .= '</figure>';
 49 
 50         }
 51 
 52     $output .= '</div>';
 53 
 54     return $output;
 55 
 56 }
