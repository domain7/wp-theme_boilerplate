 3     /**
 4      * Page title used in the content area of each page. Note this is kinda legacy.
 5      * Could be replaced with wp_title().
 6      *
 7      * @package d7
 8      * @subpackage boilerplate-theme
 9      *
10      * @return string The page title
11      *
12      */
13     function d7_get_page_title() {
14 
15         $title = '';
16 
17         if ( is_category() ) {
18             $title .= single_cat_title("", false);
19         } elseif( is_tag() ) {
20             $title .= 'Content tagged: ' . single_tag_title("", false);
21         } elseif (is_day()) {
22             $title .= 'Archive for ' . get_the_time('F jS, Y');
23         } elseif (is_month()) {
24             $title .= __('Archive for') . ' ' . get_the_time('F, Y');
25         } elseif (is_year()) {
26             $title .= __('Archive for') . ' ' . get_the_time('Y');
27         } elseif (is_search()) {
28             $title .= __('Search results');
29             if ( isset($_GET['s']) ) {
30                 $title .= ' ' . __('for') . ' <span class="search_term">&quot;' . $_GET['s'] . '&quot;</span>';
31             }
32         } elseif (is_author()) {
33             $title .= __('Author archive');
34         } elseif (isset($_GET['paged']) && !empty($_GET['paged'])) {
35             $title .= __('Archives');
36         } elseif (is_404()) {
37             $title .= __('Oh darn!');
38         } else {
39             $title .= bloginfo('name');
40         }
41 
42         return $title;
43 
44     }
